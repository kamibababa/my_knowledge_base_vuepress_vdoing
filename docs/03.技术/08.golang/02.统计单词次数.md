```go
package main

import (
	"fmt"
	"strings"
)

func main() {
	words := "this is a book that is a desk"
	strArr := strings.Split(words, " ")

	wordMap := make(map[string]int)
	for _, str := range strArr{
		if val,ok := wordMap[str]; ok{
			wordMap[str] =val+ 1
		}else{
			wordMap[str] = 1
		}

	}
	for key,val := range wordMap{
		fmt.Println(key,"-------",val)
	}
}
```
